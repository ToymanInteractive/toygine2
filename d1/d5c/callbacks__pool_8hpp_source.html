30/*!
31 \class CallbacksPool
32 \brief Template callback pool class with fixed-size callback storage.
33
34 CallbacksPool is a lightweight, high-performance callback pool class that uses a fixed-size array to store function
35 pointers. It provides a simple observer pattern implementation without dynamic memory allocation, making it suitable
36 for embedded systems, real-time applications, and performance-critical code where memory allocation overhead must be
37 minimized.
38
39 \tparam type The type of the callback parameter. Must be a copyable type.
40 \tparam allocatedSize The maximum number of callbacks that can be stored in the pool. Must be greater than zero
41 (default: \c 4).
42
43 \section features Key Features
44
45 - ‚öôÔ∏è **Zero Dynamic Allocation**: All memory is allocated on the stack at compile time
46 - üîß **ConstExpr Support**: All operations can be evaluated at compile time
47 - üõ°Ô∏è **Exception Safety**: All operations are noexcept
48 - üîó **Simple Interface**: Easy-to-use callback subscription and invocation
49 - üß¨ **Type Safety**: Strong typing with template parameters
50
51 \section usage Usage Example
52
53 \code
54 #include "core/callbacks_pool.hpp"
55
56 // Define callback function
57 void onEvent(int value) {
58 // Handle event
59 }
60
61 // Create a callback pool with default size (4 callbacks)
62 toy::CallbacksPool<int> pool;
63
64 // Subscribe to events
65 pool.add(onEvent);
66
67 // Invoke all registered callbacks
68 pool.call(42);
69
70 // Check subscriber count
71 auto count = pool.subscribersAmount();
72
73 // Unsubscribe
74 pool.remove(onEvent);
75
76 // Clear all subscriptions
77 pool.reset();
78 \endcode
79
80 \section performance Performance Characteristics
81
82 - ‚öôÔ∏è **Construction**: O(1) constant time
83 - üìù **Adding Callback**: O(n) where n is the number of registered callbacks (for duplicate check)
84 - üîó **Removing Callback**: O(n) where n is the number of registered callbacks
85 - üîç **Calling Callbacks**: O(n) where n is the number of registered callbacks
86 - üìä **Subscriber Count**: O(1) constant time (cached count)
87 - üíæ **Memory Usage**: Fixed at compile time
88 - ‚ö° **Cache Performance**: Excellent due to stack allocation and contiguous memory layout
89 - üìã **Copy Performance**: Fast due to array-based storage
90
91 \section safety Safety Guarantees
92
93 - üõ°Ô∏è **Contracts & Debug Checks**: Null pointer checks are asserted in debug; in release, violating preconditions is UB
94 - üîí **Bounds Safety**: Pool capacity is fixed at compile time; adding beyond capacity is asserted in debug
95 - üìê **Type Safety**: Template parameters ensure type-safe callback invocation
96 - ‚ö†Ô∏è **Exception Safety**: All operations are noexcept, no exceptions thrown
97
98 \note The pool does not prevent duplicate subscriptions - adding the same callback multiple times will only register
99 it once and return \c true.
100 \note When the pool is full, attempts to add new callbacks will fail (asserted in debug mode).
101
102 \warning Callbacks must remain valid for the lifetime of the CallbacksPool object. The pool does not manage callback
103 lifetime.
104*/
105template <typename type, size_t allocatedSize = 4>
106class CallbacksPool {
107public:
108 /*!
109 \brief Default constructor.
110
111 Creates an empty callback pool with all slots available.
112
113 \post The pool is empty (no callbacks registered).
114 */
115 constexpr CallbacksPool() noexcept;
116
117 /*!
118 \brief Destructor for the callback pool.
119
120 \note Since the CallbacksPool does not manage callback lifetime, no special cleanup is required.
121 */
122 constexpr ~CallbacksPool() noexcept = default;
123
124 /*!
125 \brief Adds a callback to the pool.
126
127 This method registers a callback function in the pool. If the callback is already registered, the operation succeeds
128 but does not create a duplicate entry. If the pool is full, the operation fails.
129
130 \param method A pointer to the callback function to register.
131
132 \return \c true if the callback was successfully added, \c false if the pool is full.
133
134 \pre The \a method must not be \c nullptr.
135
136 \note If the callback is already registered, the method returns \c true without adding a duplicate.
137 \note If the pool is full, the method returns \c false and asserts in debug mode.
138 */
139 constexpr bool add(void (*method)(type arg)) noexcept;
140
141 /*!
142 \brief Removes a callback from the pool.
143
144 This method unregisters a callback function from the pool. If the callback is not found, the operation has no
145 effect.
146
147 \param method A pointer to the callback function to remove.
148
149 \return \c true if the callback was found and removed, \c false otherwise.
150
151 \note Removing a non-existent callback is safe and returns \c false.
152 */
153 constexpr bool remove(void (*method)(type arg)) noexcept;
154
155 /*!
156 \brief Clears all callbacks from the pool.
157
158 This method removes all registered callbacks, resetting the pool to an empty state.
159
160 \post The pool is empty (no callbacks registered).
161 \post All callback slots are available for new subscriptions.
162 */
163 constexpr void reset() noexcept;
164
165 /*!
166 \brief Returns the number of registered callbacks.
167
168 This method returns the current number of active callbacks in the pool.
169
170 \return The number of registered callbacks.
171
172 \note The count is maintained internally for optimal performance.
173 */
174 [[nodiscard]] constexpr size_t subscribersAmount() const noexcept;
175
176 /*!
177 \brief Invokes all registered callbacks with the given parameter.
178
179 This method calls all registered callback functions with the specified parameter value. Callbacks are invoked in the
180 order they were added to the pool.
181
182 \param parameter The parameter value to pass to all registered callbacks.
183 */
184 constexpr void call(type parameter) const noexcept;
185
186private:
187 static_assert(allocatedSize > 0, "CallbacksPool capacity must be greater than zero");
188
189 //! Internal structure to store a callback function pointer.
190 struct StaticCallback {
191 void (*method)(type arg) = nullptr; //!< Pointer to the callback function.
192 };
193
194 std::array<StaticCallback, allocatedSize> _callbacks; //!< Fixed-size array storing registered callbacks.
195 size_t _subscribersCount = 0; //!< Active callbacks count.
196};
197
198} // namespace toy
