36/*!
37 \brief Converts an integer value to a string representation in a specified base.
38
39 This function converts a given integer value into its string representation in the specified base, storing the result
40 in the provided destination buffer.
41
42 \tparam type The type of the integer value to be converted.
43
44 \param dest The destination buffer where the converted string is stored.
45 \param destSize The size of the destination buffer.
46 \param value The integer value to be converted.
47 \param base The numerical base for the conversion, e.g., 10 for decimal, 16 for hexadecimal.
48
49 \return The number of characters written to the destination buffer.
50
51 \note The function assumes that the destination buffer is large enough to hold the converted string. The function does
52 not null-terminate the string. The base must be between 2 and 36 inclusive.
53*/
54template <typename type>
55constexpr std::size_t integerToSymbols(char * dest, std::size_t destSize, type value, unsigned base) {
56 static_assert(std::is_integral_v<type>, "integerToSymbols requires an integral type");
57 assert_message(dest != nullptr && destSize > 0, "The destination buffer must not be null.");
58 assert_message(base >= 2 && base <= sc_ansiDigits.size(), "The base must be between 2 and 36 inclusive.");
59
60 std::size_t index = 0;
61
62 do {
63 dest[index++] = sc_ansiDigits[value % base];
64 } while ((value /= static_cast<type>(base)) > 0 && index < destSize);
65
66 return index;
67}

69/*!
70 \brief Converts an integer value to a string representation.
71
72 This function converts a given integer value into its string representation, storing the result in the provided
73 destination buffer.
74
75 \tparam type The type of the integer value to be converted.
76
77 \param dest The destination buffer where the converted string is stored.
78 \param destSize The size of the destination buffer.
79 \param value The integer value to be converted.
80
81 \return A pointer to the destination buffer containing the converted string.
82
83 \note The function assumes that the destination buffer is large enough to hold the converted string. If the
84 destination buffer size is 1, only a null terminator is written. The function reverses the string in-place.
85*/
86template <typename type>
87inline char * itoaImplementation(char * dest, std::size_t destSize, type value) {
88 static_assert(std::is_integral_v<type> && std::is_signed_v<type>,
89 "itoaImplementation requires a signed integral type");
90 assert_message(dest != nullptr && destSize > 0, "The destination buffer must not be null.");
91 if (destSize == 1) {
92 *dest = '\0';
93
94 return dest;
95 }
96
97 // decrease dest size for '\0' symbol
98 --destSize;
99
100 std::size_t symbols = 0;
101 const bool valueNegative = (value < 0);
102 using unsigned_type = std::make_unsigned_t<type>;
103 if (valueNegative) {
104 symbols = integerToSymbols(dest, destSize, static_cast<unsigned_type>(-(value + 1)) + 1U, 10);
105 } else {
106 symbols = integerToSymbols(dest, destSize, static_cast<unsigned_type>(value), 10);
107 }
108
109 if (valueNegative && symbols < destSize)
110 dest[symbols++] = '-';
111
112 dest[symbols] = '\0';
113 reverseString(dest, symbols);
114
115 return dest;
116}

118/*!
119 \brief Converts an unsigned integer value to a string representation in a specified base.
120
121 This function converts a given unsigned integer value into its string representation in the specified numerical base,
122 storing the result in the provided destination buffer. The resulting string is reversed in place and null-terminated.
123
124 \tparam type The type of the unsigned integer value to be converted.
125
126 \param dest The destination buffer where the converted string is stored.
127 \param destSize The size of the destination buffer.
128 \param value The unsigned integer value to be converted.
129 \param base The numerical base for the conversion, e.g., 10 for decimal, 16 for hexadecimal.
130
131 \return A pointer to the destination buffer containing the converted string.
132
133 \note The function assumes that the destination buffer is large enough to hold the converted string. If the
134 destination buffer size is 1, only a null terminator is written. The base must be between 2 and 36 inclusive.
135*/
136template <typename type>
137inline char * utoaImplementation(char * dest, std::size_t destSize, type value, unsigned base) {
138 static_assert(std::is_unsigned_v<type>, "utoaImplementation requires an unsigned integer type");
139 assert_message(dest != nullptr && destSize > 0, "The destination buffer must not be null.");
140 if (destSize == 1) {
141 *dest = '\0';
142
143 return dest;
144 }
145
146 // decrease dest size for '\0' symbol
147 --destSize;
148
149 auto symbols = integerToSymbols(dest, destSize, value, base);
150
151 dest[symbols] = '\0';
152 reverseString(dest, symbols);
153
154 return dest;
155}
