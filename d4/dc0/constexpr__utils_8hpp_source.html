30/*!
31 \brief Compares two C strings lexicographically.
32
33 This function performs a lexicographic comparison between two C strings. The comparison is performed character by
34 character using the character's numeric value. This function is designed to be constexpr-compatible and provides the
35 same behavior as std::strcmp but can be evaluated at compile time.
36
37 \param lhs The left-hand side C string to compare.
38 \param rhs The right-hand side C string to compare.
39
40 \return A negative value if \a lhs is lexicographically less than \a rhs, zero if they are equal,
41 or a positive value if \a lhs is lexicographically greater than \a rhs.
42
43 \pre The \a lhs pointer must not be null.
44 \pre The \a rhs pointer must not be null.
45
46 \note The comparison is case-sensitive.
47 \note The comparison stops at the first character that differs between the strings.
48 \note If one string is a prefix of another, the shorter string is considered lexicographically smaller.
49 \note Return values are -1, 0, or 1 based on lexicographic order using unsigned char comparison; this differs from
50 std::strcmp, which returns the arithmetic difference.
51 \note This function is constexpr-compatible and can be used in compile-time contexts.
52
53 \see std::strcmp
54*/
55[[nodiscard]] constexpr int cstrcmp(const char * lhs, const char * rhs) noexcept;

57/*!
58 \brief Finds the first occurrence of a character in a C string.
59
60 This function searches for the first occurrence of the character \a ch within the string \a str. The search is
61 performed character by character from the beginning of the string and returns a pointer to the first occurrence, or
62 nullptr if the character is not found. This function is designed to be constexpr-compatible and provides the same
63 behavior as std::strchr but can be evaluated at compile time.
64
65 \param str The C string to search in.
66 \param ch The character to search for.
67
68 \return A pointer to the first occurrence of \a ch in \a str, or nullptr if not found.
69
70 \pre The \a str pointer must not be null.
71
72 \note The search is case-sensitive.
73 \note The function searches from the beginning of \a str and returns the first match.
74 \note This function is constexpr-compatible and can be used in compile-time contexts.
75
76 \see std::strchr
77*/
78[[nodiscard]] constexpr const char * cstrchr(const char * str, int ch) noexcept;

80/*!
81 \brief Finds the first occurrence of any character from a set in a C string.
82
83 This function searches for the first occurrence of any character from the \a accept string within the \a str string.
84 The search is performed character by character from the beginning and returns a pointer to the first character found
85 that matches any character in the accept set, or nullptr if no character from the accept set is found. This function
86 is designed to be constexpr-compatible and provides the same behavior as std::strpbrk but can be evaluated at compile
87 time.
88
89 \param str The C string to search in. Must be null-terminated.
90 \param accept The C string containing the set of characters to search for. Must be null-terminated.
91
92 \return A pointer to the first character in \a str that matches any character in \a accept, or nullptr if no
93 character from the accept set is found.
94
95 \pre The \a str pointer must not be null.
96 \pre The \a accept pointer must not be null.
97
98 \note The search is case-sensitive.
99 \note The function searches from the beginning of \a str and returns the first match.
100 \note This function is constexpr-compatible and can be used in compile-time contexts.
101
102 \see std::strpbrk
103*/
104[[nodiscard]] constexpr const char * cstrpbrk(const char * str, const char * accept) noexcept;
105
106/*!
107 \brief Finds the first occurrence of a substring in a C string.
108
109 This function searches for the first occurrence of the \a needle string within the \a haystack string. The search is
110 performed character by character from the beginning and returns a pointer to the beginning of the first occurrence, or
111 nullptr if the substring is not found. This function is designed to be constexpr-compatible and provides the same
112 behavior as std::strstr but can be evaluated at compile time.
113
114 \param haystack The C string to search in.
115 \param needle The C string to search for.
116
117 \return A pointer to the first occurrence of \a needle in \a haystack, or nullptr if not found. If \a needle is an
118 empty string, returns \a haystack.
119
120 \pre The \a haystack pointer must not be null.
121 \pre The \a needle pointer must not be null.
122
123 \note The search is case-sensitive.
124 \note If \a needle is an empty string, the function returns \a haystack immediately.
125 \note The function searches from the beginning of \a haystack and returns the first match.
126 \note This function is constexpr-compatible and can be used in compile-time contexts.
127
128 \see std::strstr
129*/
130[[nodiscard]] constexpr const char * cstrstr(const char * haystack, const char * needle) noexcept;
