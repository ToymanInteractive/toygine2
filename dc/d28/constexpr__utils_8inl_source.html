30constexpr int cstrcmp(const char * lhs, const char * rhs) noexcept {
31 while (*lhs && (*lhs == *rhs)) {
32 ++lhs;
33 ++rhs;
34 }
35
36 if (*lhs == *rhs)
37 return 0;
38
39 return (static_cast<unsigned char>(*lhs) < static_cast<unsigned char>(*rhs)) ? -1 : 1;
40}

42constexpr const char * cstrchr(const char * str, int ch) noexcept {
43 const auto target = static_cast<unsigned char>(ch);
44
45 for (; *str; ++str) {
46 if (static_cast<unsigned char>(*str) == target)
47 return str;
48 }
49
50 return (static_cast<unsigned char>(*str) == target) ? str : nullptr;
51}

53constexpr const char * cstrpbrk(const char * str, const char * accept) noexcept {
54 if (str == nullptr || accept == nullptr)
55 return nullptr;
56
57 for (; *str != '\0'; ++str) {
58 for (const char * a = accept; *a != '\0'; ++a) {
59 if (*str == *a)
60 return str;
61 }
62 }
63
64 return nullptr;
65}

67constexpr const char * cstrstr(const char * haystack, const char * needle) noexcept {
68 if (!*needle)
69 return haystack;
70
71 const char first = *needle;
72 for (const char * h = haystack; *h; ++h) {
73 if (*h != first)
74 continue;
75
76 const char * h_it = h;
77 const char * n_it = needle;
78
79 while (*h_it && *n_it && *h_it == *n_it) {
80 ++h_it;
81 ++n_it;
82 }
83
84 if (!*n_it)
85 return h;
86 }
87
88 return nullptr;
89}
