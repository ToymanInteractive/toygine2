32template <typename StringType>
33class OStringStream {
34public:
35 /*!
36 \brief Default constructor.
37
38 Constructs an empty OStringStream with default-initialized string storage.
39
40 \post The stream contains an empty string.
41 */
42 constexpr OStringStream() noexcept = default;
43
44 /*!
45 \brief Constructs an OStringStream from a string-like object.
46
47 This constructor initializes the stream by copying the content from a string-like object. The source object must
48 satisfy the \ref toy::StringLike concept.
49
50 \tparam SourceStringType The type of the source string. Must satisfy the \ref toy::StringLike concept.
51
52 \param string The source string-like object to copy content from.
53
54 \pre The \a string must be valid and properly initialized.
55 \pre The \a string size must not exceed the capacity of StringType.
56
57 \post The stream contains a copy of the source string content.
58
59 \note The constructor performs a deep copy of the string content.
60 \note The source string type can be different from StringType as long as both satisfy \ref toy::StringLike.
61 */
62 template <StringLike SourceStringType>
63 explicit constexpr OStringStream(const SourceStringType & string) noexcept;
64
65 /*!
66 \brief Copy assignment operator.
67
68 Assigns the contents of another OStringStream to this stream. The assignment performs a deep copy of the underlying
69 string.
70
71 \param other The source OStringStream to copy from.
72
73 \return A reference to this OStringStream after assignment.
74
75 \pre The \a other stream must be valid and properly initialized.
76
77 \post This stream contains the same content as the source \a other stream.
78 \post The underlying string is a copy of the source string.
79
80 \note Self-assignment is handled correctly and safely.
81 */
82 constexpr OStringStream & operator=(const OStringStream & other) noexcept = default;
83
84 /*!
85 \brief Move assignment operator.
86
87 Assigns the contents of another OStringStream to this stream using move semantics. The source stream is left in a
88 valid but unspecified state.
89
90 \param other The source OStringStream to move from.
91
92 \return A reference to this OStringStream after assignment.
93
94 \pre The \a other stream must be valid and properly initialized.
95
96 \post This stream contains the same content as the source \a other stream.
97 \post The source \a other stream is left in a valid but unspecified state.
98
99 \note Self-assignment is handled correctly and safely.
100 */
101 constexpr OStringStream & operator=(OStringStream && other) noexcept = default;
102
103 /*!
104 \brief Swaps the contents of this stream with another stream.
105
106 This method exchanges the underlying string storage between this stream and the \a other stream. Both streams must
107 have the same StringType template parameter.
108
109 \param other The stream to swap contents with.
110
111 \post The contents of this stream and \a other are exchanged.
112 \post Both streams remain valid after the swap operation.
113
114 \note Self-swap is safe and has no effect.
115 \note The operation is noexcept and provides strong exception guarantee.
116 \note This method is useful for efficient content exchange without copying.
117 */
118 constexpr void swap(OStringStream & other) noexcept;
119
120 /*!
121 \brief Returns a const reference to the underlying string.
122
123 This method provides read-only access to the internal string storage.
124
125 \return A const reference to the underlying string.
126
127 \note The returned reference is valid for the lifetime of the OStringStream object.
128 */
129 [[nodiscard]] constexpr const StringType & str() const noexcept;
130
131private:
132 StringType _string;
133};
134
135} // namespace toy
