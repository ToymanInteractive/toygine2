279constexpr size_t CStringView::_find_raw(size_t position, const char * data, size_t dataSize) const noexcept {
280 const auto stringViewSize = size();
281
282 if (position > stringViewSize)
283 return npos;
284
285 if (dataSize == 0)
286 return position;
287 else if (dataSize > stringViewSize - position)
288 return npos;
289
290 const char * occurrence;
291
292 if consteval {
293 occurrence = dataSize == 1 ? cstrchr(_data + position, data[0]) : cstrstr(_data + position, data);
294 } else {
295 occurrence = dataSize == 1
296 ? static_cast<const char *>(std::memchr(_data + position, data[0], stringViewSize - position))
297 : std::strstr(_data + position, data);
298 }
299
300 return occurrence != nullptr ? static_cast<size_t>(occurrence - _data) : npos;
301}

303constexpr size_t CStringView::_rfind_raw(size_t position, const char * data, size_t dataSize) const noexcept {
304 const auto stringViewSize = size();
305
306 if (dataSize == 0)
307 return std::min(position, stringViewSize);
308 else if (dataSize > stringViewSize)
309 return npos;
310
311 if (position == npos)
312 position = stringViewSize - dataSize;
313 else if (position > stringViewSize - dataSize)
314 return npos;
315
316 for (size_t i = 0; i <= position; ++i) {
317 const auto offset = position - i;
318
319 const auto found = char_traits<char>::compare(_data + offset, data, dataSize) == 0;
320 if (found)
321 return offset;
322 }
323
324 return npos;
325}

328 const auto stringViewSize = size();

344constexpr size_t CStringView::_find_first_not_of_raw(size_t position, const char * data,
345 size_t dataSize) const noexcept {
346 const auto stringViewSize = size();
347
348 if (position >= stringViewSize)
349 return npos;
350
351 if (dataSize == 0)
352 return position;
353
354 if (dataSize == 1) {
355 const auto exclude = data[0];
356 for (auto i = position; i < stringViewSize; ++i) {
357 if (_data[i] != exclude)
358 return i;
359 }
360 } else {
361 std::array<bool, 256> excludedChars{};
362
363 for (size_t i = 0; i < dataSize; ++i) {
364 excludedChars[static_cast<unsigned char>(data[i])] = true;
365 }
366
367 for (auto i = position; i < stringViewSize; ++i) {
368 if (!excludedChars[static_cast<unsigned char>(_data[i])])
369 return i;
370 }
371 }
372
373 return npos;
374}

377 if (dataSize == 0 || empty())
378 return npos;
379
380 const auto stringViewSize = size();
381
382 if (position == npos)
383 position = stringViewSize - 1;
384 else if (position >= stringViewSize)
385 return npos;

387 if (dataSize == 1) {

389 for (size_t i = 0; i <= position; ++i) {
