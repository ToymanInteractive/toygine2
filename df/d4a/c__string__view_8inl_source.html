168constexpr std::size_t CStringView::_find_raw(std::size_t position, const char * data,
169 std::size_t dataSize) const noexcept {
170 const auto stringViewSize = size();
171
172 if (position > stringViewSize)
173 return npos;
174
175 if (dataSize == 0)
176 return position;
177 else if (dataSize > stringViewSize - position)
178 return npos;
179
180 const char * occurrence;
181
182 if consteval {
183 occurrence = dataSize == 1 ? cstrchr(_data + position, data[0]) : cstrstr(_data + position, data);
184 } else {
185 occurrence = dataSize == 1
186 ? static_cast<const char *>(std::memchr(_data + position, data[0], stringViewSize - position))
187 : std::strstr(_data + position, data);
188 }
189
190 return occurrence != nullptr ? static_cast<std::size_t>(occurrence - _data) : npos;
191}

193constexpr std::size_t CStringView::_rfind_raw(std::size_t position, const char * data,
194 std::size_t dataSize) const noexcept {
195 const auto stringViewSize = size();
196
197 if (dataSize == 0)
198 return std::min(position, stringViewSize);
199 else if (dataSize > stringViewSize)
200 return npos;
201
202 if (position == npos)
203 position = stringViewSize - dataSize;
204 else if (position > stringViewSize - dataSize)
205 return npos;
206
207 for (std::size_t i = 0; i <= position; ++i) {
208 const auto offset = position - i;
209
210 bool found;
211
212 if consteval {
213 found = std::equal(_data + offset, _data + offset + dataSize, data);
214 } else {
215 found = std::memcmp(_data + offset, data, dataSize) == 0;
216 }
217
218 if (found)
219 return offset;
220 }
221
222 return npos;
223}

225constexpr std::size_t CStringView::_find_first_of_raw(std::size_t position, const char * data,
226 std::size_t dataSize) const noexcept {
227 const auto stringViewSize = size();
228
229 if (position >= stringViewSize || dataSize == 0)
230 return npos;
231
232 const char * occurrence;
233
234 if consteval {
235 occurrence = dataSize == 1 ? cstrchr(_data + position, data[0]) : cstrpbrk(_data + position, data);
236 } else {
237 occurrence = dataSize == 1 ? std::strchr(_data + position, data[0]) : std::strpbrk(_data + position, data);
238 }
239
240 return occurrence != nullptr ? static_cast<std::size_t>(occurrence - _data) : npos;
241}
