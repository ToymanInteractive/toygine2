28namespace toygine {

30template <std::size_t allocatedSize>
31class FixString {
32public:
33 /*!
34 \brief Default constructor.
35
36 \post The string is empty and ready for use.
37 */
38 constexpr inline FixString() noexcept;
39
40 /*!
41 \brief Destructor for the string.
42
43 This destructor cleans up the string.
44
45 \note Since the FixString does not manage dynamic memory, no special cleanup is required.
46 */
47 inline ~FixString() noexcept;
48
49 /*!
50 \brief Constructs a copy of \a string.
51
52 This constructor initializes a string by copying the content from another \a string.
53
54 \param string The source string to copy content from.
55
56 \pre The source \a string must be valid and properly initialized.
57
58 \post The new string has the same size as the source \a string.
59 */
60 constexpr inline FixString(const FixString<allocatedSize> & string) noexcept;
61
62 /*!
63 \brief Constructs a string initialized with a StringLike object.
64
65 This constructor initializes a string by copying the content from a StringLike object.
66
67 \tparam stringType The type of the source string. Must satisfy the StringLike concept.
68
69 \param string The source StringLike object to copy content from.
70
71 \pre The source StringLike object must be valid and properly initialized.
72 \pre The source StringLike object size must not exceed the allocated capacity.
73
74 \post The new string is created with the contents of the source StringLike object.
75 */
76 template <StringLike stringType>
77 constexpr inline explicit FixString(const stringType & string) noexcept;
78
79 /*!
80 \brief Constructs a string initialized with the C \a string.
81
82 This constructor initializes a string by copying the content from the C \a string. The constructor automatically
83 determines the length of the source string and copies the content up to the allocated capacity.
84
85 \param string The source C string to copy content from.
86
87 \pre The source \a string must not be null.
88
89 \post The new string is created with the contents of the source C \a string.
90 */
91 constexpr explicit inline FixString(const char * string) noexcept;
92
93 /*!
94 \brief Constructs a string of the given \a count of \a character.
95
96 This constructor initializes a string with a \a character repeated the given \a count of times.
97
98 \param character The character to repeat.
99 \param count The number of times to repeat the \a character (default: 1).
100
101 \pre The \a count must not exceed the allocated size.
102
103 \post The new string is created with the contents of the \a character repeated the given \a count of times.
104 \post The string size equals the \a count.
105
106 \note This is useful for creating strings with repeated patterns or filling strings with specific characters.
107 */
108 constexpr inline explicit FixString(char character, std::size_t count = 1) noexcept;

110 /*!
