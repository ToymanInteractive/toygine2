30/*!
31 \brief Compares two C strings lexicographically.
32
33 This function performs a lexicographic comparison between two C strings. The comparison is performed character by
34 character using the character's numeric value. This function is designed to be constexpr-compatible and provides the
35 same behavior as std::strcmp but can be evaluated at compile time.
36
37 \param lhs The left-hand side C string to compare.
38 \param rhs The right-hand side C string to compare.
39
40 \return A negative value if \a lhs is lexicographically less than \a rhs, zero if they are equal,
41 or a positive value if \a lhs is lexicographically greater than \a rhs.
42
43 \pre The \a lhs pointer must not be null.
44 \pre The \a rhs pointer must not be null.
45
46 \note The comparison is case-sensitive.
47 \note The comparison stops at the first character that differs between the strings.
48 \note If one string is a prefix of another, the shorter string is considered lexicographically smaller.
49 \note Return values are -1, 0, or 1 based on lexicographic order using unsigned char comparison; this differs from
50 std::strcmp, which returns the arithmetic difference.
51 \note This function is constexpr-compatible and can be used in compile-time contexts.
52
53 \see std::strcmp
54*/
55[[nodiscard]] constexpr int cstrcmp(const char * lhs, const char * rhs) noexcept;

57/*!
58 \brief Finds the first occurrence of a character in a C string.
59
60 This function searches for the first occurrence of the character \a ch within the string \a str. The search is
61 performed character by character from the beginning of the string and returns a pointer to the first occurrence, or
62 nullptr if the character is not found. This function is designed to be constexpr-compatible and provides the same
63 behavior as std::strchr but can be evaluated at compile time.
64
65 \param str The C string to search in.
66 \param ch The character to search for.
67
68 \return A pointer to the first occurrence of \a ch in \a str, or nullptr if not found.
69
70 \pre The \a str pointer must not be null.
71
72 \note The search is case-sensitive.
73 \note The function searches from the beginning of \a str and returns the first match.
74 \note This function is constexpr-compatible and can be used in compile-time contexts.
75
76 \see std::strchr
77*/
78[[nodiscard]] constexpr const char * cstrchr(const char * str, int ch) noexcept;

80/*!
81 \brief Finds the first occurrence of a substring in a C string.
82
83 This function searches for the first occurrence of the \a needle string within the \a haystack string. The search is
84 performed character by character from the beginning and returns a pointer to the beginning of the first occurrence, or
85 nullptr if the substring is not found. This function is designed to be constexpr-compatible and provides the same
86 behavior as std::strstr but can be evaluated at compile time.
87
88 \param haystack The C string to search in.
89 \param needle The C string to search for.
90
91 \return A pointer to the first occurrence of \a needle in \a haystack, or nullptr if not found. If \a needle is an
92 empty string, returns \a haystack.
93
94 \pre The \a haystack pointer must not be null.
95 \pre The \a needle pointer must not be null.
96
97 \note The search is case-sensitive.
98 \note If \a needle is an empty string, the function returns \a haystack immediately.
99 \note The function searches from the beginning of \a haystack and returns the first match.
100 \note This function is constexpr-compatible and can be used in compile-time contexts.
101
102 \see std::strstr
103*/
104[[nodiscard]] constexpr const char * cstrstr(const char * haystack, const char * needle) noexcept;
