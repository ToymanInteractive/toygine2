30template <typename type, std::size_t allocatedSize>
31class FixedVector {
32public:
33 using value_type = type; //!< Type of elements stored in the vector.
34 using size_type = std::size_t; //!< Type used for vector size and capacity.
35 using difference_type = std::ptrdiff_t; //!< Type used for pointer differences.
36 using reference = value_type &; //!< Reference to vector element.
37 using const_reference = const value_type &; //!< Const reference to vector element.
38 using pointer = type *; //!< Pointer to vector element.
39 using const_pointer = const type *; //!< Const pointer to vector element.
40
41 /*!
42 \brief Default constructor.
43
44 \post The vector is empty and ready for use.
45 */
46 constexpr FixedVector() noexcept;
47
48 /*!
49 \brief Destructor for the vector.
50
51 This destructor cleans up the vector.
52
53 \note Since the FixedVector manages its own elements, proper cleanup is performed.
54 */
55 constexpr ~FixedVector() noexcept;
56
57 /*!
58 \brief Constructs a FixedVector with \a count default-initialized elements.
59
60 This constructor creates a FixedVector containing \a count elements, each initialized with the default value of
61 type.
62
63 \param count The number of elements to create. Must not exceed the allocated capacity.
64
65 \pre The \a count must not exceed the allocated capacity.
66
67 \post The vector contains \a count elements, each initialized with the default value of type.
68 \post The vector size is equal to \a count.
69 */
70 explicit FixedVector(size_type count);
71
72 /*!
73 \brief Constructs a FixedVector with \a count elements initialized to \a value.
74
75 This constructor creates a FixedVector containing \a count elements, each initialized with the specified \a value.
76
77 \param count The number of elements to create. Must not exceed the allocated capacity.
78 \param value The value to initialize all elements with.
79
80 \pre The \a count must not exceed the allocated capacity.
81 \pre The \a value must be a valid instance of type.
82
83 \post The vector contains \a count elements, each initialized with \a value.
84 \post The vector size is equal to \a count.
85 */
86 FixedVector(size_type count, const type & value);
87
88 /*!
89 \brief Constructs a copy of \a other FixedVector.
90
91 This constructor initializes a FixedVector by copying the content from another FixedVector of the same type and
92 capacity.
93
94 \param other The source FixedVector to copy content from.
95
96 \pre The source \a other must be valid and properly initialized.
97
98 \post The new vector has the same size and content as the source \a other.
99 */
100 FixedVector(const FixedVector<type, allocatedSize> & other) noexcept;
101
102 /*!
103 \brief Constructs a FixedVector by copying from another FixedVector with different capacity.
104
105 This constructor initializes a FixedVector by copying the content from another FixedVector of the same type
106 but potentially different capacity.
107
108 \tparam allocatedSize2 The capacity of the source FixedVector. Must be greater than zero.
109
110 \param other The source FixedVector to copy content from.
111
112 \pre The source \a other must be valid and properly initialized.
113 \pre The source \a other size must not exceed the allocated capacity of this vector.
114
115 \post The new vector has the same size and content as the source \a other.
116 */
117 template <std::size_t allocatedSize2>
118 FixedVector(const FixedVector<type, allocatedSize2> & other) noexcept;
119
120 /*!
121 \brief Constructs a FixedVector by moving from another FixedVector.
122
123 This constructor initializes a FixedVector by moving the content from another FixedVector of the same type and
124 capacity. The source vector is left in a valid but unspecified state.
125
126 \param other The source FixedVector to move content from.
127
128 \pre The source \a other must be valid and properly initialized.
129
130 \post The new vector has the same size and content as the source \a other.
131 \post The source \a other is left in a valid but unspecified state.
132 */
133 FixedVector(FixedVector<type, allocatedSize> && other) noexcept;
134
135 /*!
136 \brief Constructs a FixedVector by moving from another FixedVector with different capacity.
137
138 This constructor initializes a FixedVector by moving the content from another FixedVector of the same type
139 but potentially different capacity. The source vector is left in a valid but unspecified state.
140
141 \tparam allocatedSize2 The capacity of the source FixedVector. Must be greater than zero.
142
143 \param other The source FixedVector to move content from.
144
145 \pre The source \a other must be valid and properly initialized.
146 \pre The source \a other size must not exceed the allocated capacity of this vector.
147
148 \post The new vector has the same size and content as the source \a other.
149 \post The source \a other is left in a valid but unspecified state.
150 */
151 template <std::size_t allocatedSize2>
152 FixedVector(FixedVector<type, allocatedSize2> && other) noexcept;
153
154 /*!
155 \brief Constructs a FixedVector from an initializer list.
156
157 This constructor initializes a FixedVector with the elements from the provided initializer list.
158
159 \param init The initializer list containing the elements to initialize the vector with.
160
161 \pre The \a init size must not exceed the allocated capacity.
162 \pre All elements in \a init must be valid instances of type.
163
164 \post The new vector has the same size and content as the source \a init.
165 */
166 explicit FixedVector(std::initializer_list<type> init) noexcept;
167
168 /*!
169 \brief Copy assigns other FixedVector to this FixedVector.
170
171 This operator assigns the contents of another FixedVector with the same capacity to this vector.
172
173 \param other The FixedVector to copy content from.
174
175 \return A reference to this FixedVector after assignment.
176
177 \pre The \a other vector must be valid and properly constructed.
178
179 \post This vector has the same size and content as the source \a other vector.
180 \post All elements are properly copied using copy construction.
181 \post All previous elements are properly destroyed.
182
183 \note Self-assignment is handled correctly and safely.
184 */
185 FixedVector<type, allocatedSize> & operator=(const FixedVector<type, allocatedSize> & other) noexcept;
186
187 /*!
188 \brief Copy assigns FixedVector with different capacity to this FixedVector.
189
190 This operator assigns the contents of another FixedVector with different capacity to this vector.
191
192 \tparam allocatedSize2 The capacity of the source FixedVector.
193
194 \param other The FixedVector to copy content from.
195
196 \return A reference to this FixedVector after assignment.
197
198 \pre The \a other vector must be valid and properly constructed.
199 \pre The \a other vector size must not exceed this vector's capacity.
200
201 \post This vector has the same size and content as the source \a other vector.
202 \post All elements are properly copied using copy construction.
203 \post All previous elements are properly destroyed.
204 */
205 template <std::size_t allocatedSize2>
206 FixedVector<type, allocatedSize> & operator=(const FixedVector<type, allocatedSize2> & other) noexcept;
207
208 /*!
209 \brief Move assigns other FixedVector to this FixedVector.
210
211 This operator assigns the contents of another FixedVector with the same capacity to this vector using move
212 semantics.
213
214 \param other The FixedVector to move content from.
215
216 \return A reference to this FixedVector after assignment.
217
218 \pre The \a other vector must be valid and properly constructed.
219
220 \post This vector has the same size and content as the source \a other vector.
221 \post All elements are moved using move construction.
222 \post All previous elements are properly destroyed.
223 \post The source vector is left in a valid but unspecified state.
224
225 \note Self-assignment is handled correctly and safely.
226 */
227 FixedVector<type, allocatedSize> & operator=(FixedVector<type, allocatedSize> && other) noexcept;
228
229 /*!
230 \brief Move assigns FixedVector with different capacity to this FixedVector.
231
232 This operator assigns the contents of another FixedVector with different capacity to this vector using move
233 semantics.
234
235 \tparam allocatedSize2 The capacity of the source FixedVector.
236
237 \param other The FixedVector to move content from.
238
239 \return A reference to this FixedVector after assignment.
240
241 \pre The \a other vector must be valid and properly constructed.
242 \pre The \a other vector size must not exceed this vector's capacity.
243
244 \post This vector has the same size and content as the source \a other vector.
245 \post All elements are moved using move construction.
246 \post All previous elements are properly destroyed.
247 \post The source vector is left in a valid but unspecified state.
248 */
249 template <std::size_t allocatedSize2>
250 FixedVector<type, allocatedSize> & operator=(FixedVector<type, allocatedSize2> && other) noexcept;
251
252 /*!
253 \brief Assigns initializer list to this FixedVector.
254
255 This operator assigns the contents of an initializer list to this vector.
256
257 \param init The initializer list containing the elements to assign.
258
259 \return A reference to this FixedVector after assignment.
260
261 \pre The \a init size must not exceed the allocated capacity.
262 \pre All elements in \a init must be valid instances of type.
263
264 \post This vector has the same size and content as the source \a init.
265 \post All elements are properly copied using copy construction.
266 \post All previous elements are properly destroyed.
267
268 \note The operator handles empty initializer lists correctly.
269 */
270 FixedVector<type, allocatedSize> & operator=(std::initializer_list<type> init) noexcept;
271
272 // temporary
273
274 constexpr std::size_t size() const noexcept;
275 constexpr std::size_t capacity() const noexcept;
276
277 constexpr void push_back(const type & val) noexcept;
278
279 constexpr void clear() noexcept;
280
281 constexpr type * end() noexcept;
282
283 constexpr const type & operator[](size_t index) const noexcept;
284
285private:
286 static_assert(allocatedSize > 0, "FixedVector capacity must be greater than zero.");
287
288 /// Internal buffer for storing elements.
289 std::byte _data[allocatedSize * sizeof(type)];
290
291 /// Current number of elements in the vector.
292 std::size_t _size;
293};
