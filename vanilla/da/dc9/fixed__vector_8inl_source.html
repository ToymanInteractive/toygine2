80template <typename type, std::size_t allocatedSize>
81constexpr void FixedVector<type, allocatedSize>::push_back(const type & val) noexcept {
82 assert(_size < allocatedSize);
83
84 if (_size < allocatedSize) {
85 PLACEMENT_NEW(end()) type(val);
