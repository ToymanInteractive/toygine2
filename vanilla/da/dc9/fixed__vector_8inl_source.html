40template <typename type, std::size_t allocatedSize>
41inline FixedVector<type, allocatedSize>::FixedVector(std::size_t count)
42 : _data()
43 , _size(count) {

132template <typename type, std::size_t allocatedSize>
133constexpr void FixedVector<type, allocatedSize>::push_back(const type & val) noexcept {
134 assert(_size < allocatedSize);
135
136 if (_size < allocatedSize) {
137 std::construct_at(end(), type(val));
138 ++_size;
139 }
140}

142template <typename type, std::size_t allocatedSize>
143constexpr type * FixedVector<type, allocatedSize>::end() noexcept {
144 return reinterpret_cast<type *>(_data) + _size;
145}

147template <typename type, std::size_t allocatedSize>

149 assert(index < _size);
