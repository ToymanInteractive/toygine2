54/*!
55 \def assert(expression)
56 \brief Debug assertion macro for runtime \a expression checking.
57
58 This macro provides runtime assertion checking in debug builds. It evaluates the given \a expression and triggers an
59 assertion failure if the \a expression is false. In release builds, this macro expands to nothing.
60
61 \param expression The boolean expression to evaluate. Must be convertible to bool.
62
63 \note This macro is only active in debug builds (_DEBUG defined).
64 \note In release builds, this macro expands to ((void)0) and has no effect.
65 \note Assertion failures will call toy::assertion::assertion with file, function, and line information.
66 \note The \a expression is evaluated only once and should not have side effects.
67
68 \warning Do not rely on the \a expression being evaluated in release builds.
69 \warning The \a expression should not have side effects as it may not be evaluated in release builds.
70
71 \see assert_message
72 \see toy::assertion::assertion
73*/
74#define assert(expression)
75 do {
76 if (std::is_constant_evaluated()) {
77 toy::assertion::constexpr_assert(expression, #expression);
78 } else if (!(expression)) {
79 toy::assertion::assertion(#expression, nullptr, __FILE__, __FUNC_SIGNATURE__, __LINE__);
80 }
81 } while (0)
82
83/*!
84 \def assert_message(expression, message)
85 \brief Debug assertion macro with custom \a message for runtime \a expression checking.
86
87 This macro provides runtime assertion checking in debug builds with a custom error \a message. It evaluates the given
88 \a expression and triggers an assertion failure with the provided \a message if the \a expression is false. In release
89 builds, this macro expands to nothing.
90
91 \param expression The boolean expression to evaluate. Must be convertible to bool.
92 \param message A custom error message to display on assertion failure. Must be a C string literal.
93
94 \note This macro is only active in debug builds (_DEBUG defined).
95 \note In release builds, this macro expands to ((void)0) and has no effect.
96 \note Assertion failures will call toy::assertion::assertion with file, function, line, and custom \a message.
97 \note The \a expression is evaluated only once and should not have side effects.
98 \note The \a message parameter is passed as-is to the assertion handler.
99
100 \warning Do not rely on the \a expression being evaluated in release builds.
101 \warning The \a expression should not have side effects as it may not be evaluated in release builds.
102 \warning The \a message should be a C string literal for optimal performance.
103
104 \see assert
105 \see toy::assertion::assertion
106*/
107#define assert_message(expression, message)
108 do {
109 if (std::is_constant_evaluated()) {
110 toy::assertion::constexpr_assert(expression, message);
111 } else if (!(expression)) {
112 toy::assertion::assertion(#expression, message, __FILE__, __FUNC_SIGNATURE__, __LINE__);
113 }
114 } while (0)
115
116#else // _DEBUG

118/*!
119 \def assert(expression)
120 \brief Release build assertion macro (no-op).
121
122 In release builds, this macro expands to nothing and has no effect. The \a expression is not evaluated.
123
124 \param expression The boolean expression (ignored in release builds).
125
126 \note This macro is only defined in release builds (!_DEBUG).
127 \note The \a expression parameter is ignored and not evaluated.
128 \note This ensures zero overhead in release builds.
129
130 \see assert_message

143
144 \note This macro is only defined in release builds (!_DEBUG).

146 \note This ensures zero overhead in release builds.
147
