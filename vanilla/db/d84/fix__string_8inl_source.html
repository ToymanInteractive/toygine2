316template <std::size_t allocatedSize>
317constexpr inline FixString<allocatedSize> & FixString<allocatedSize>::insert(std::size_t index, char character,
318 std::size_t count) noexcept {
319 if (count == 0)
320 return *this;
321
322 assert_message(index <= _size, "Index must not exceed string size");
323 assert_message(_size + count < allocatedSize, "Inserted characters must fit in capacity");
324
325 // if inserting at the end, just append
326 if (index == _size) {
327 std::memset(_data + index, character, count);
328 _size += count;
329 _data[_size] = '\0';
330 } else {
331 std::memmove(_data + index + count, _data + index, _size - index + 1);
332 std::memset(_data + index, character, count);
333 _size += count;
334 }
335
336 return *this;
337}

339template <std::size_t allocatedSize>
340constexpr inline FixString<allocatedSize> & FixString<allocatedSize>::erase(std::size_t offset,
341 std::size_t count) noexcept {
342 assert_message(offset <= _size, "Offset must not exceed string size");
343
344 if (count == npos)
345 count = _size - offset;
346
347 if (count == 0)
348 return *this;
349
350 assert_message(offset + count <= _size, "Erase range must be within string bounds");
351
352 _size -= count;
353 std::memmove(_data + offset, _data + offset + count, _size - offset + 1);
354
355 return *this;
356}

374template <std::size_t allocatedSize>
375constexpr inline void FixString<allocatedSize>::utf8_pop_back() noexcept {
376 assert_message(_size > 0, "String must not be empty for utf8_pop_back");
377
378 while (_size > 0) {
379 --_size;
380 const auto byte = static_cast<unsigned char>(_data[_size]);
381 _data[_size] = '\0';
382 if ((byte & 0xC0) != 0x80)
383 break;
384 }

499template <std::size_t allocatedSize>
500constexpr inline FixString<allocatedSize> & FixString<allocatedSize>::replace(std::size_t pos, std::size_t count,
501 char character,
502 std::size_t charactersCount) noexcept {
503 if (count == 0 && charactersCount == 0)
504 return *this;
505
506 assert_message(pos <= _size, "Position must be within string bounds");
507 assert_message(pos + count <= _size, "Replacement range must be within string bounds");
508
509 const auto newSize = _size - count + charactersCount;
510
511 assert_message(newSize < allocatedSize, "Replacement result must fit in capacity");
512
513 // If sizes are equal, no need to shift data
514 if (count == charactersCount) {
515 std::memset(_data + pos, character, charactersCount);
516 } else {
517 // If replacing at the end, no need to shift
518 if (pos + count == _size) {
519 std::memset(_data + pos, character, charactersCount);
520 _data[pos + charactersCount] = '\0';
521 } else {
522 std::memmove(_data + pos + charactersCount, _data + pos + count, _size - pos - count + 1);
523 std::memset(_data + pos, character, charactersCount);
524 }
525
526 _size = newSize;
527 }
528
529 return *this;
530}

532template <std::size_t allocatedSize>
533constexpr inline std::size_t FixString<allocatedSize>::copy(char * dest, std::size_t count,
534 std::size_t pos) const noexcept {
535 assert_message(pos <= _size, "Position must be within string bounds");
536 assert_message(dest != nullptr, "Destination pointer must not be null");
537 assert_message((dest < _data) || (dest >= (_data + allocatedSize)),
538 "Destination buffer must not overlap internal storage");
539
540 if (count == npos || pos + count > _size)
541 count = _size - pos;
542
543 std::memcpy(dest, _data + pos, count);
544
545 return count;
546}

548template <std::size_t allocatedSize>
549constexpr inline void FixString<allocatedSize>::swap(FixString<allocatedSize> & string) noexcept {
550 if (this == &string)
551 return;
552
553 char tempData[allocatedSize];
554 std::memcpy(tempData, _data, _size + 1);
555 std::memcpy(_data, string._data, string._size + 1);
556 std::memcpy(string._data, tempData, _size + 1);
557
558 std::swap(_size, string._size);
559}

850template <std::size_t allocatedSize>
851constexpr inline void FixString<allocatedSize>::_insert_raw(std::size_t position, const char * data,
852 std::size_t dataSize) noexcept {
853 if (dataSize == 0)
854 return;
855
856 assert_message(((data + dataSize) < _data) || (data >= (_data + allocatedSize)),
857 "Source data pointer must not point into _data buffer");
858
859 assert_message(position <= _size, "Index must not exceed string size");
860 assert_message(_size + dataSize < allocatedSize, "Inserted string must fit in capacity");
861
862 // If inserting at the end, just append
863 if (position == _size) {
864 std::memcpy(_data + _size, data, dataSize + 1);
865 } else {
866 std::memmove(_data + position + dataSize, _data + position, _size - position + 1);
867 std::memcpy(_data + position, data, dataSize);
868 }
869
870 _size += dataSize;
871}

886template <std::size_t allocatedSize>
887constexpr inline void FixString<allocatedSize>::_replace_raw(std::size_t position, std::size_t oldCount,
888 const char * data, std::size_t dataSize) noexcept {
889 if (oldCount == 0 && dataSize == 0)
890 return;
891
892 assert_message(((data + dataSize) < _data) || (data >= (_data + allocatedSize)),
893 "Source data pointer must not point into _data buffer");
894
895 assert_message(position <= _size, "Position must be within string bounds");
896 assert_message(position + oldCount <= _size, "Replacement range must be within string bounds");
897
898 // If sizes are equal, no need to shift data
899 if (oldCount == dataSize) {
900 std::memcpy(_data + position, data, dataSize);
901 return;
902 }
903
904 assert_message((_size - oldCount + dataSize) < allocatedSize, "Replacement result must fit in capacity");
905
906 // If replacing at the end, no need to shift
907 if (position + oldCount == _size) {
908 std::memcpy(_data + position, data, dataSize + 1);
909 } else {
910 std::memmove(_data + position + dataSize, _data + position + oldCount, _size - position - oldCount + 1);
911 std::memcpy(_data + position, data, dataSize);
912 }
913
914 _size = _size - oldCount + dataSize;
915}
