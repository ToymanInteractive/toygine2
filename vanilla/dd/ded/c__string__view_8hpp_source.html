30class CStringView {

32 /*!
33 \brief Default constructor.

36 */
37 constexpr CStringView() noexcept;

39 /*!
40 \brief Constructs a copy of \a string view.
41
42 This constructor initializes a string view by copying the C string pointer from another \a string view.
43
44 \param string The source string view to copy C string pointer from.
45
46 \pre The source \a string view must be valid and properly initialized.
47 */
48 constexpr CStringView(const CStringView & string) noexcept = default;
49
50 /*!
51 \brief Constructs a string view from a C \a string.
52
53 Creates a new string view that wraps the provided C \a string pointer.
54
55 \param string The C string to wrap.
56
57 \pre The source C \a string must not be null.
58 \pre The source C \a string must be null-terminated.
59 */
60 constexpr CStringView(const char * string) noexcept;

62 /*!
63 \brief Destructor for the string view.
64
65 \note Since the Destructor does not manage dynamic memory, no special cleanup is required.
66 */

69 /*!
70 \brief Copy assigns other \a string view to this string view.

72 This operator assigns the C string pointer of another \a string view to this string view.

76 \return A reference to this string view after assignment.

78 constexpr CStringView & operator=(const CStringView & string) noexcept = default;

80 /*!
81 \brief Copy assigns the C \a string to this string view.

83 This operator assigns a new C \a string to this string view.

85 \param string The C string to assign.

94 /*!
95 \brief Copy assigns other \a string view to this string view.

110 \brief Copy assigns the C \a string to this string view.

127 /*!
128 \brief Access a character in the string view at a given \a offset.
129
130 This method provides read-only access to a character at the specified \a offset within the string view.
131
132 \param offset The offset of the character to access in the string view.
133
134 \return A const reference to the character at the specified \a offset.
135
136 \pre The \a offset should be less than the current string view size.

150 \param offset The offset of the character to access in the string view.
151
152 \return A const reference to the character at the specified \a offset.

182 [[nodiscard]] constexpr const char & back() const noexcept;

185 \brief Returns a constant pointer to the data of the string view.
186
187 This method returns a constant pointer to the internal character array that stores the string view data. The

189
190 \return A constant pointer to the internal character array.
191
192 \note The returned pointer points to a null-terminated character array.

196

198 \brief Returns a constant pointer to the C string representation of this string view.
199
200 This method returns a constant pointer to the C string representation of this string view. The returned pointer
201 provides read-only access to the string view contents and can be used with C functions that require a char pointer.
202
203 \return A constant pointer to the C string representation of this string view.
204
205 \note The returned pointer points to a null-terminated string view.
206 \note The returned pointer is read-only and cannot modify the string view contents.
207 \note This method is equivalent to data() const method.
208 */
209 [[nodiscard]] constexpr const char * c_str() const noexcept;
210
211 /*!
212 \brief Checks if the string view is empty.
213
214 This method checks if the string view is empty, i.e. its size is zero. An empty string view contains no characters
215 and has a length of zero.
216
217 \return True if the string view is empty, false otherwise.
218
219 \note An empty string view has size zero.
220 \note An empty string view still contains a null terminator.
221 \note This method is equivalent to the expression: ```size() == 0```.
222 */
223 [[nodiscard]] constexpr bool empty() const noexcept;
224
225 /*!
226 \brief Returns the size of the string view.
227
228 This method returns the current number of characters in the string view, excluding the terminating null character.
229 The size represents the actual length of the string view content.
230
231 \return The number of characters in the string view, excluding the terminating null character.
232
233 \note This method is equivalent to length() method.
234 */
235 [[nodiscard]] constexpr std::size_t size() const noexcept;
236
237 /*!
238 \brief Returns the size of the Unicode string in UTF-8 encoding.
239
240 This method returns the number of Unicode characters in the UTF-8 encoded string, excluding the terminating null
241 character. For ASCII strings, this value equals the size() method. For UTF-8 encoded strings, this method counts the
242 number of Unicode characters rather than bytes. The method validates UTF-8 encoding and counts only complete,
243 well-formed Unicode characters.
244
245 \return The number of Unicode characters in the string view, excluding the terminating null character.
246
247 \note For ASCII strings, utf8_size() equals size().
248 \note For UTF-8 strings, utf8_size() may be less than size().
249 \note Invalid UTF-8 sequences are handled gracefully and may affect the count.
250 \note This method is useful for internationalization and text processing applications.
251 */
252 [[nodiscard]] inline std::size_t utf8_size() const noexcept;
253
254 /*!
255 \brief Returns the size of the string view.
256
257 This method returns the current number of characters in the string view, excluding the terminating null character.
258 The length represents the actual length of the string view content.
259
260 \return The number of characters in the string view, excluding the terminating null character.
261
262 \note This method is equivalent to size() method.
263
264 \see size()
265 */
266 [[nodiscard]] constexpr std::size_t length() const noexcept;
267
268 /*!
269 \brief Returns the maximum observable size of this string view.
270
271 This method returns the maximum possible size for the string view. The maximum size represents the size of the
272 allocated buffer.
273
274 \return The maximum number of characters in the string view, excluding the terminating null character.
275
276 \note This method is equivalent to size() method.
277 */
278 [[nodiscard]] constexpr std::size_t max_size() const noexcept;
279
280 /*!
281 \brief Returns the capacity of the string view.
282
283 This method returns the capacity of the allocated buffer for the string view.
284
285 \return The capacity of the string view in characters, excluding the terminating null character.
286
287 \note This method is equivalent to size() method.
288 */
289 [[nodiscard]] constexpr std::size_t capacity() const noexcept;
290
291 /*!
292 \brief Clears the contents by detaching from the current C string.
293
294 This method resets this view to point at the internal empty sentinel. The previously referenced character buffer is
295 not modified or freed.
296
297 \note No allocation or deallocation occurs.
298 */
299 constexpr void clear() noexcept;
300
301private:
302 /// Pointer to the wrapped C string
303 const char * _data;
304
305 /// Static empty C string used as default value for null pointers
306 static constexpr char _emptyString[]{'\0'};
307};
