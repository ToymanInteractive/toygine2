30class CStringView {

37 constexpr CStringView() noexcept;

39 /*!
40 \brief Constructs a copy of \a string view.
41
42 This constructor initializes a string view by copying the C string pointer from another \a string view.
43

47 */
48 constexpr CStringView(const CStringView & string) noexcept = default;

50 /*!
51 \brief Constructs a string view from a C \a string.

58 \pre The source C \a string must be null-terminated.

60 constexpr CStringView(const char * string) noexcept;

62 /*!
63 \brief Destructor for the string view.

65 \note Since the Destructor does not manage dynamic memory, no special cleanup is required.

69 /*!
70 \brief Copy assigns other \a string view to this string view.

78 constexpr CStringView & operator=(const CStringView & string) noexcept = default;

80 /*!

127 /*!
128 \brief Access a character in the string view at a given \a offset.
129
130 This method provides read-only access to a character at the specified \a offset within the string view.
131

211 /*!
212 \brief Checks if the string view is empty.
213
214 This method checks if the string view is empty, i.e. its size is zero. An empty string view contains no characters

216
217 \return True if the string view is empty, false otherwise.
218

220 \note An empty string view still contains a null terminator.

222 */
223 [[nodiscard]] constexpr bool empty() const noexcept;

225 /*!
226 \brief Returns the size of the string view.
227
228 This method returns the current number of characters in the string view, excluding the terminating null character.
229 The size represents the actual length of the string view content.
230
231 \return The number of characters in the string view, excluding the terminating null character.
232
233 \note This method is equivalent to length() method.
234 */
235 [[nodiscard]] constexpr std::size_t size() const noexcept;
236
237 /*!
238 \brief Returns the size of the Unicode string in UTF-8 encoding.
239
240 This method returns the number of Unicode characters in the UTF-8 encoded string, excluding the terminating null
241 character. For ASCII strings, this value equals the size() method. For UTF-8 encoded strings, this method counts the
242 number of Unicode characters rather than bytes. The method validates UTF-8 encoding and counts only complete,
243 well-formed Unicode characters.
244
245 \return The number of Unicode characters in the string view, excluding the terminating null character.
246
247 \note For ASCII strings, utf8_size() equals size().
248 \note For UTF-8 strings, utf8_size() may be less than size().
249 \note Invalid UTF-8 sequences are handled gracefully and may affect the count.
250 \note This method is useful for internationalization and text processing applications.
251 */
252 [[nodiscard]] inline std::size_t utf8_size() const noexcept;

254 /*!
255 \brief Returns the size of the string view.

259
260 \return The number of characters in the string view, excluding the terminating null character.

262 \note This method is equivalent to size() method.
263
264 \see size()
265 */
266 [[nodiscard]] constexpr std::size_t length() const noexcept;

268 /*!
269 \brief Returns the maximum observable size of this string view.

301 /*!
302 \brief Swaps the contents of this string view with another \a string view.
